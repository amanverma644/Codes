
	//Fast Modulo Expnentiation using bitmasks

/*
Given two number a and b. Calculate a^b 
*/

//Time Complexity: O(logN)
#include <bits/stdc++.h>
using namespace std;

#define ll long long
#define MOD 10000007

int fastModExp(ll a, ll b) {
	ll result = 1;
	while (b > 0) {
		if (b & 1) {
			result = (result * a) % MOD;
		}
		a = (a * a) % MOD;
		b = b >> 1;
	}
	return result;
}

int main() {
	int a = 5, b = 2;
	cout << fastModExp(a, b) << endl;

	return 0;
}
