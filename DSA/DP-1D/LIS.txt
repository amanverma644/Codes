
	//LIS (Longest Increasing Subsequence) compelete code

#include <bits/stdc++.h>
using namespace std;

int lis(vector<int> arr){
    int n=arr.size();
    vector<int> dp(n,1);
    int lis=1;
    for(int i=1;i<n;i++){
        for(int j=0;j<i;j++){
            if(arr[j]<arr[i]){
                dp[i]=max(dp[i],1+dp[j]);
                lis=max(lis,dp[i]);
            }
        }
    }
    return lis;
}

int main() {
    vector<int> arr={5,4,10,8,30,100,2};
    cout<<lis(arr)<<endl;
    return 0;
}
